# AIEffect - 数据库设计文档

## 1. 概述

### 1.1 数据库选型

- **数据库**: SQLite 3
- **选择理由**:
  - 轻量级，无需独立服务器
  - 单文件存储，便于备份和迁移
  - 适合桌面应用的本地数据存储
  - 支持标准 SQL，开发成本低

### 1.2 数据库文件位置

```
data/
├── aieffect.db          # 主数据库文件
├── aieffect.db-wal      # WAL 模式日志文件
├── aieffect.db-shm      # 共享内存文件
└── backups/             # 数据库备份目录
    └── aieffect_20240115_103000.db
```

### 1.3 设计原则

- 使用外键约束保证数据完整性
- 时间戳字段统一使用 UTC 时间
- 敏感数据 (API Key) 加密存储
- 预留扩展字段 (extra_data JSON)

### 2.1 配置表 (config)

存储应用全局配置，采用键值对形式便于扩展。

```sql
CREATE TABLE config (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    key TEXT NOT NULL UNIQUE,           -- 配置键名
    value TEXT,                          -- 配置值 (JSON 字符串)
    category TEXT DEFAULT 'general',     -- 配置分类: general/api/tts/subtitle/memory/ports
    is_encrypted BOOLEAN DEFAULT 0,      -- 是否加密存储
    description TEXT,                    -- 配置说明
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建索引
CREATE INDEX idx_config_category ON config(category);
CREATE INDEX idx_config_key ON config(key);
```

### 2.2 API 服务商预设表 (provider_presets)

存储内置和自定义的 API 服务商预设。

```sql
CREATE TABLE provider_presets (
    id TEXT PRIMARY KEY,                 -- 预设 ID (如: deepseek, custom_xxx)
    name TEXT NOT NULL,                  -- 显示名称
    icon TEXT DEFAULT '⭐',              -- 图标 (emoji)
    api_url TEXT NOT NULL,               -- API 基础 URL
    api_key TEXT,                        -- API 密钥 (加密存储)

    doc_url TEXT,                        -- 文档地址
    is_custom BOOLEAN DEFAULT 0,         -- 是否自定义预设
    is_builtin BOOLEAN DEFAULT 0,        -- 是否内置预设
    sort_order INTEGER DEFAULT 0,        -- 排序顺序
    extra_data TEXT,                     -- 扩展数据 (JSON)
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建索引
CREATE INDEX idx_provider_presets_custom ON provider_presets(is_custom);
CREATE INDEX idx_provider_presets_builtin ON provider_presets(is_builtin);
```

### 2.4 语音角色表 (voice_characters)

存储语音合成的角色配置。

```sql
CREATE TABLE voice_characters (
```

### 2.8 角色管理表 (characters)

存储「存档角色/游戏角色」管理信息，包括角色名称、所属存档、AI 灵魂、AI 声优、头像、Token 消耗、聊天次数、会话压缩设置、交互操作等。

```sql
CREATE TABLE characters (
    id TEXT PRIMARY KEY,                 -- 角色ID (建议 ULID/UUID)
    name TEXT NOT NULL,                  -- 角色名称
    save_id TEXT,                        -- 所属存档ID (可为空：未绑定存档/仅作为模板角色)
    ai_soul TEXT,                        -- AI灵魂 (角色性格描述)
    ai_voice TEXT,                       -- AI声优 (语音角色配置)
    avatar_url TEXT,                     -- 头像地址/路径 (建议存 file path，不建议存 dataURL)
    token_usage INTEGER DEFAULT 0,       -- Token消耗总量
    chat_count INTEGER DEFAULT 0,        -- 聊天次数
    compression_enabled BOOLEAN DEFAULT 0, -- 是否启用会话压缩
    interaction_ops TEXT DEFAULT '[]',   -- 交互操作列表 (JSON数组: ["对话", "任务", "交易", ...])
    extra_data TEXT,                     -- 扩展数据 (JSON，存放导入原始 payload 等)
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建索引
CREATE INDEX idx_characters_save_id ON characters(save_id);
CREATE INDEX idx_characters_name ON characters(name);
CREATE INDEX idx_characters_created ON characters(created_at);
```

## 6. Python 数据库操作示例

### 6.1 数据库连接封装

```python
# backend/src/core/database.py
import sqlite3
import json
from pathlib import Path
from contextlib import contextmanager
from typing import Optional, Any, List, Dict

DATABASE_PATH = Path("./data/aieffect.db")

@contextmanager
def get_db_connection():
    """获取数据库连接上下文管理器"""
    DATABASE_PATH.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(DATABASE_PATH)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA foreign_keys = ON")
    try:
        yield conn
        conn.commit()
    except Exception as e:
        conn.rollback()
        raise e
    finally:
        conn.close()

class ConfigManager:
    """配置管理器"""

    @staticmethod
    def get(key: str, default: Any = None) -> Any:
        with get_db_connection() as conn:
            cursor = conn.execute(
                "SELECT value FROM config WHERE key = ?",
                (key,)
            )
            row = cursor.fetchone()
            if row:
                return json.loads(row['value'])
            return default

    @staticmethod
    def set(key: str, value: Any, category: str = 'general'):
        with get_db_connection() as conn:
            conn.execute(
                """INSERT INTO config (key, value, category)
                   VALUES (?, ?, ?)
                   ON CONFLICT(key) DO UPDATE SET
                   value = excluded.value,
                   updated_at = CURRENT_TIMESTAMP""",
                (key, json.dumps(value), category)
            )

    @staticmethod
    def get_all(category: Optional[str] = None) -> Dict[str, Any]:
        with get_db_connection() as conn:
            if category:
                cursor = conn.execute(
                    "SELECT key, value FROM config WHERE category = ?",
                    (category,)
                )
            else:
                cursor = conn.execute("SELECT key, value FROM config")

            return {row['key']: json.loads(row['value']) for row in cursor}

class LogManager:
    """日志管理器"""

    @staticmethod
    def add(level: str, module: str, message: str, metadata: Dict = None):
        with get_db_connection() as conn:
            conn.execute(
                """INSERT INTO system_logs
                   (level, module, message, metadata)
                   VALUES (?, ?, ?, ?)""",
                (level, module, message, json.dumps(metadata) if metadata else None)
            )

    @staticmethod
    def query(
        level: Optional[str] = None,
        module: Optional[str] = None,
        search: Optional[str] = None,
        page: int = 1,
        page_size: int = 50
    ) -> Dict:
        with get_db_connection() as conn:
            conditions = []
            params = []

            if level:
                conditions.append("level = ?")
                params.append(level)
            if module:
                conditions.append("module = ?")
                params.append(module)
            if search:
                conditions.append("message LIKE ?")
                params.append(f"%{search}%")

            where_clause = "WHERE " + " AND ".join(conditions) if conditions else ""

            # 获取总数
            count_sql = f"SELECT COUNT(*) FROM system_logs {where_clause}"
            total = conn.execute(count_sql, params).fetchone()[0]

            # 获取分页数据
            offset = (page - 1) * page_size
            data_sql = f"""SELECT * FROM system_logs
                          {where_clause}
                          ORDER BY timestamp DESC
                          LIMIT ? OFFSET ?"""
            cursor = conn.execute(data_sql, params + [page_size, offset])

            items = [dict(row) for row in cursor]

            return {
                "items": items,
                "total": total,
                "page": page,
                "page_size": page_size,
                "total_pages": (total + page_size - 1) // page_size
            }
```

### 7.2 迁移脚本示例

```python
# backend/src/core/migrations.py
MIGRATIONS = [
    {
        "version": "001",
        "name": "initial_schema",
        "sql": """
            -- 创建初始表结构
            CREATE TABLE config (...);
            CREATE TABLE provider_presets (...);
            -- ...
        """
    },
    {
        "version": "002",
        "name": "add_voice_characters",
        "sql": """
            CREATE TABLE voice_characters (...);
            CREATE TABLE character_emotions (...);
        """
    }
]

def run_migrations():
    """执行数据库迁移"""
    with get_db_connection() as conn:
        # 确保迁移表存在
        conn.execute("""
            CREATE TABLE IF NOT EXISTS db_migrations (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                version TEXT NOT NULL UNIQUE,
                name TEXT NOT NULL,
                applied_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                checksum TEXT
            )
        """)

        # 获取已应用的迁移
        applied = set(row['version'] for row in
                     conn.execute("SELECT version FROM db_migrations"))

        # 应用新迁移
        for migration in MIGRATIONS:
            if migration['version'] not in applied:
                conn.executescript(migration['sql'])
                conn.execute(
                    "INSERT INTO db_migrations (version, name) VALUES (?, ?)",
                    (migration['version'], migration['name'])
                )
                print(f"Applied migration: {migration['version']} - {migration['name']}")
```

```sql
    session_count INTEGER DEFAULT 0,     -- 活跃会话数
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建索引
CREATE INDEX idx_websocket_logs_connection ON websocket_connection_logs(connection_id);
CREATE INDEX idx_websocket_logs_created ON websocket_connection_logs(created_at DESC);
CREATE INDEX idx_websocket_logs_event ON websocket_connection_logs(event_type);
```
