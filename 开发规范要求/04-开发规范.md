# AIEffect - 开发规范文档

## 1. Git 工作流

### 1.1 分支策略

| 分支名        | 用途     | 说明                 |
| ------------- | -------- | -------------------- |
| `main`        | 主分支   | 稳定版本，仅用于发布 |
| `develop`     | 开发分支 | 日常开发集成         |
| `feature/xxx` | 功能分支 | 新功能开发           |
| `bugfix/xxx`  | 修复分支 | Bug 修复             |
| `hotfix/xxx`  | 紧急修复 | 生产环境紧急修复     |

### 1.2 Commit 规范

采用 Conventional Commits 规范：

```
<type>(<scope>): <subject>

<body>

<footer>
```

**Type 类型**：

- `feat`: 新功能
- `fix`: 修复 Bug
- `docs`: 文档更新
- `style`: 代码格式调整
- `refactor`: 重构
- `perf`: 性能优化
- `test`: 测试相关
- `chore`: 构建/工具链相关

**Scope 范围**：

- `frontend`: 前端相关
- `api-gateway`: API 网关服务
- `websocket`: WebSocket 服务
- `log`: 日志服务
- `shared`: 共享组件
- `deps`: 依赖更新

**示例**：

```
feat(api-gateway): add GPT-SoVITS engine support

- Implement GPT-SoVITS adapter
- Add voice configuration API
- Add unit tests

Closes #123
```

### 2.5 API 调用规范

// utils/api.ts
const API_BASE_URL = import.meta.env.VITE_API_URL || "http://localhost:8501";

export async function fetchApi<T>(
  endpoint: string,
  options?: RequestInit
): Promise<T> {
  const response = await fetch(`${API_BASE_URL}${endpoint}`, {
    headers: {
      "Content-Type": "application/json",
    },
    ...options,
  });

  if (!response.ok) {
    throw new Error(`API Error: ${response.statusText}`);
  }

  return response.json();
}

// 使用示例

### 2.6 WebSocket 使用规范

```typescript
// hooks/useWebSocket.ts
import { useEffect, useRef, useCallback } from "react";

export function useWebSocket(url: string) {
  const ws = useRef<WebSocket | null>(null);

  useEffect(() => {
    ws.current = new WebSocket(url);

    ws.current.onopen = () => {
      console.log("WebSocket connected");
    };

    ws.current.onclose = () => {
      console.log("WebSocket disconnected");
    };

    return () => {
      ws.current?.close();
    };
  }, [url]);

  const sendMessage = useCallback((data: unknown) => {
    if (ws.current?.readyState === WebSocket.OPEN) {
      ws.current.send(JSON.stringify(data));
    }
  }, []);

  return { sendMessage };
}
```

### 3.3 FastAPI 路由规范

```python
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel

router = APIRouter(prefix="/api/tts", tags=["tts"])

class SynthesizeRequest(BaseModel):
    text: str
    voice_id: str
    speed: float = 1.0

class SynthesizeResponse(BaseModel):
    audio_url: str
    duration: float

@router.post("/synthesize", response_model=SynthesizeResponse)
async def synthesize_speech(
    request: SynthesizeRequest,
    tts_service: TTSService = Depends(get_tts_service),
):
    try:
        result = await tts_service.synthesize(
            text=request.text,
            voice_id=request.voice_id,
            speed=request.speed,
        )
        return SynthesizeResponse(
            audio_url=result.audio_url,
            duration=result.duration,
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

### 4.2 端口管理代码

```python
# shared/python/shared_core/port_manager.py
import socket
from typing import Optional

class PortManager:
    """端口管理器，处理端口分配和冲突"""

    DEFAULT_PORTS = {
        'frontend': 8500,
        'api_gateway': 8501,
        'websocket': 8502,
        'subtitle': 8503,
        'tts': 8504,
        'log_service': 8505,
    }

    PORT_RANGE = (8500, 8505)

    @classmethod
    def is_port_available(cls, port: int, host: str = "0.0.0.0") -> bool:
        """检查端口是否可用"""
        try:
            with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
                s.bind((host, port))
                return True
        except OSError:
            return False

    @classmethod
    def find_available_port(cls, preferred_port: int) -> int:
        """查找可用端口，优先使用指定端口"""
        if cls.is_port_available(preferred_port):
            return preferred_port

        # 在范围内查找下一个可用端口
        for port in range(preferred_port + 1, cls.PORT_RANGE[1] + 1):
            if cls.is_port_available(port):
                return port

        raise RuntimeError(f"无法在范围 {cls.PORT_RANGE} 内找到可用端口")
```

import httpx
from fastapi import HTTPException

async def call_log_service(log_data: dict):
    """调用日志服务"""
    settings = get_settings()
    async with httpx.AsyncClient() as client:
        try:
            response = await client.post(
                f"{settings.log_service_url}/api/logs",
                json=log_data,
                timeout=5.0
            )
            response.raise_for_status()
            return response.json()
        except httpx.RequestError as e:

## 10. 性能优化指南

### 10.1 前端优化

- 使用 React.memo 避免不必要重渲染
- PixiJS 画布使用对象池减少 GC
- 路由懒加载
- 图片懒加载
- 使用 CDN 加载第三方库

### 10.2 后端优化

- 使用异步 I/O (async/await)
- 数据库查询优化
- 响应压缩 (gzip)
- 缓存常用数据
- 连接池管理

### 10.3 Docker 优化

- 使用多阶段构建减小镜像体积
- 合理使用缓存层
- 使用 .dockerignore 排除不需要的文件
- 非 root 用户运行容器

## 11. 安全规范
