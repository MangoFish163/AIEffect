### 2.2 端口冲突处理策略

```python
# backend/src/core/port_manager.py
import socket
from typing import Optional

class PortManager:
    """端口管理器，处理端口分配和冲突"""
    
    DEFAULT_PORTS = {
        'api': 8500,
        'ollama_proxy': 8501,
        'websocket': 8502,
        'subtitle': 8503,
        'tts': 8504,
        'log': 8505,
    }
    
    PORT_RANGE = (8500, 8600)
    
    @classmethod
    def is_port_available(cls, port: int) -> bool:
        """检查端口是否可用"""
        try:
            with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
                s.bind(('127.0.0.1', port))
                return True
        except OSError:
            return False
    
    @classmethod
    def find_available_port(cls, preferred_port: int) -> int:
        """查找可用端口，优先使用指定端口"""
        if cls.is_port_available(preferred_port):
            return preferred_port
        
        # 在范围内查找下一个可用端口
        for port in range(preferred_port + 1, cls.PORT_RANGE[1] + 1):
            if cls.is_port_available(port):
                return port
        
        raise RuntimeError(f"无法在范围 {cls.PORT_RANGE} 内找到可用端口")
    
    @classmethod
    def get_service_port(cls, service_name: str, config: Optional[dict] = None) -> int:
        """获取服务端口，支持配置覆盖"""
        preferred = config.get(f'{service_name}_port') if config else None
        preferred = preferred or cls.DEFAULT_PORTS.get(service_name, 8500)
        return cls.find_available_port(preferred)
```

### 3.3 打包脚本

```python
# scripts/build_backend.py
#!/usr/bin/env python3
"""
后端打包脚本
"""
import os
import sys
import shutil
import subprocess
from pathlib import Path

# 端口配置检查
REQUIRED_PORTS = [8500, 8501, 8502, 8503, 8504, 8505]

def check_ports():
    """检查端口配置是否正确"""
    config_file = Path("backend/src/core/config.py")
    content = config_file.read_text(encoding='utf-8')
    
    for port in REQUIRED_PORTS:
        if str(port) not in content:
            print(f"⚠️ 警告: 端口 {port} 未在配置中找到")
            return False
    
    print("✅ 端口配置检查通过")
    return True

def build_windows():
    """构建 Windows 版本"""
    print("🔨 构建 Windows 版本...")
    
    cmd = [
        sys.executable, "-m", "nuitka",
        "--standalone",
        "--enable-plugin=anti-bloat",
        "--include-package=fastapi",
        "--include-package=uvicorn",
        "--include-package=pydantic",
        "--windows-disable-console",
        "--windows-icon-from-ico=resources/icon.ico",
        "--output-dir=dist/windows",
        "backend/src/main.py"
    ]
    
    subprocess.run(cmd, check=True)
    print("✅ Windows 构建完成")

def build_macos():
    """构建 macOS 版本"""
    print("🔨 构建 macOS 版本...")
    
    cmd = [
        sys.executable, "-m", "nuitka",
        "--standalone",
        "--macos-app-icon=resources/icon.icns",
        "--macos-app-name=AIEffect-Backend",
        "--output-dir=dist/macos",
        "backend/src/main.py"
    ]
    
    subprocess.run(cmd, check=True)
    print("✅ macOS 构建完成")

def build_linux():
    """构建 Linux 版本"""
    print("🔨 构建 Linux 版本...")
    
    cmd = [
        sys.executable, "-m", "nuitka",
        "--standalone",
        "--linux-icon=resources/icon.png",
        "--output-dir=dist/linux",
        "backend/src/main.py"
    ]
    
    subprocess.run(cmd, check=True)
    print("✅ Linux 构建完成")

if __name__ == "__main__":
    if not check_ports():
        sys.exit(1)
    
    platform = sys.platform
    if platform == "win32":
        build_windows()
    elif platform == "darwin":
        build_macos()
    else:
        build_linux()
```

### 4.2 主进程端口管理

```typescript
// frontend/electron/main.ts
import { app, ipcMain } from 'electron';
import { spawn, ChildProcess } from 'child_process';
import * as net from 'net';

// 端口配置
const PORT_CONFIG = {
  API: 8500,
  OLLAMA_PROXY: 8501,
  WEBSOCKET: 8502,
  SUBTITLE: 8503,
  TTS: 8504,
  LOG: 8505,
};

class PortManager {
  private usedPorts: Map<string, number> = new Map();
  
  async findAvailablePort(preferredPort: number): Promise<number> {
    return new Promise((resolve, reject) => {
      const server = net.createServer();
      
      server.listen(preferredPort, '127.0.0.1', () => {
        const port = (server.address() as net.AddressInfo).port;
        server.close(() => resolve(port));
      });
      
      server.on('error', (err: any) => {
        if (err.code === 'EADDRINUSE') {
          // 端口被占用，尝试下一个
          this.findAvailablePort(preferredPort + 1).then(resolve).catch(reject);
        } else {
          reject(err);
        }
      });
    });
  }
  
  async allocatePorts(): Promise<Record<string, number>> {
    const ports: Record<string, number> = {};
    
    for (const [name, defaultPort] of Object.entries(PORT_CONFIG)) {
      ports[name] = await this.findAvailablePort(defaultPort);
      this.usedPorts.set(name, ports[name]);
      console.log(`✅ 分配端口 ${name}: ${ports[name]}`);
    }
    
    return ports;
  }
  
  getPort(serviceName: string): number | undefined {
    return this.usedPorts.get(serviceName);
  }
}

// 后端进程管理
class BackendManager {
  private process: ChildProcess | null = null;
  private portManager = new PortManager();
  
  async start() {
    const ports = await this.portManager.allocatePorts();
    
    // 设置环境变量
    const env = {
      ...process.env,
      AIEFFECT_API_PORT: String(ports.API),
      AIEFFECT_OLLAMA_PROXY_PORT: String(ports.OLLAMA_PROXY),
      AIEFFECT_WS_PORT: String(ports.WEBSOCKET),
      AIEFFECT_SUBTITLE_PORT: String(ports.SUBTITLE),
      AIEFFECT_TTS_PORT: String(ports.TTS),
      AIEFFECT_LOG_PORT: String(ports.LOG),
    };
    
    // 启动后端
    const backendPath = app.isPackaged
      ? path.join(process.resourcesPath, 'backend', 'aieffect-backend')
      : path.join(__dirname, '../../backend/dist/aieffect-backend');
    
    this.process = spawn(backendPath, [], {
      env,
      detached: false,
    });
    
    this.process.stdout?.on('data', (data) => {
      console.log(`[Backend] ${data}`);
    });
    
    this.process.stderr?.on('data', (data) => {
      console.error(`[Backend Error] ${data}`);
    });
    
    return ports;
  }
  
  stop() {
    if (this.process) {
      this.process.kill();
      this.process = null;
    }
  }
}

export const backendManager = new BackendManager();
```

## 6. 自动更新配置

### 6.1 更新服务器配置

```typescript
// frontend/electron/updater.ts
import { autoUpdater } from 'electron-updater';
import { dialog, ipcMain } from 'electron';
import * as path from 'path';

// 配置自动更新
autoUpdater.autoDownload = false;
autoUpdater.autoInstallOnAppQuit = true;

// 检查更新
export function checkForUpdates() {
  autoUpdater.checkForUpdatesAndNotify();
}

// 监听更新事件
autoUpdater.on('checking-for-update', () => {
  console.log('🔍 检查更新...');
});

autoUpdater.on('update-available', (info) => {
  console.log('📦 发现新版本:', info.version);
  
  dialog.showMessageBox({
    type: 'info',
    title: '发现新版本',
    message: `AIEffect ${info.version} 可用`,
    detail: '是否现在下载更新？',
    buttons: ['立即下载', '稍后提醒'],
    defaultId: 0,
  }).then((result) => {
    if (result.response === 0) {
      autoUpdater.downloadUpdate();
    }
  });
});

autoUpdater.on('update-not-available', () => {
  console.log('✅ 已是最新版本');
});

autoUpdater.on('download-progress', (progress) => {
  const percent = Math.round(progress.percent);
  console.log(`⬇️ 下载进度: ${percent}%`);
  
  // 发送进度到渲染进程
  mainWindow?.webContents.send('update-progress', {
    percent,
    speed: progress.bytesPerSecond,
  });
});

autoUpdater.on('update-downloaded', (info) => {
  console.log('✅ 更新下载完成');
  
  dialog.showMessageBox({
    type: 'info',
    title: '更新已就绪',
    message: '新版本已下载完成',
    detail: '重启应用以安装更新',
    buttons: ['立即重启', '稍后重启'],
    defaultId: 0,
  }).then((result) => {
    if (result.response === 0) {
      autoUpdater.quitAndInstall();
    }
  });
});

autoUpdater.on('error', (error) => {
  console.error('❌ 更新错误:', error);
});
```
